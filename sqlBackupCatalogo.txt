CREATE TABLE Materiali (
	idMateriale int primary key AUTO_INCREMENT,
   	nomeMateriale VARCHAR(255) NOT NULL,
    luogoAcquisto VARCHAR(255),
    costoMateriale DECIMAL(8,2)
);

CREATE TABLE ComposizioneOggetto (
  idComp int primary key AUTO_INCREMENT,
  idMat1 int,
  idMat2 INT,
  idMat3 INT,
  idMat4 INT,
  idMat5 INT,
  FOREIGN KEY (idMat1) REFERENCES Materiali(idMateriale),
  FOREIGN KEY (idMat2) REFERENCES Materiali(idMateriale),
  FOREIGN KEY (idMat3) REFERENCES Materiali(idMateriale),
  FOREIGN KEY (idMat4) REFERENCES Materiali(idMateriale),
  FOREIGN KEY (idMat5) REFERENCES Materiali(idMateriale)
);
